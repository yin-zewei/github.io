POJ1328 Rader Installation

发表于 2020-02-01 | 分类于 算法
字数统计 308 | 阅读时长 1

Description
对每个岛屿，得到一个雷达的活动范围,然后根据活动范围的左界进行排序,再通过贪心进行求解，如果下一个范围左界小于当前右界，那么说明重合,不用添加雷达，如果大于当前右界，那么需要添加一个雷达，并且更新右界;

```c++

#include<cmath>
#include<cstdio>
#include<algorithm>
#define MAX 1005
using namespace std;
struct note{                //雷达范围
    double left,right;
}b[MAX];
bool cmp(note a,note b)      //比较函数
{
    if(a.right==b.right) return a.left>b.left; 
        else
    return a.right<b.right;
}
int acount,n; 
void search(double maxn)     //贪心代码
{
    int i;
    for(i=1;i<n;i++)
        if(b[i].left>maxn)
        {
            maxn=b[i].right;        
            acount++;
        }
}
int main()
{
    double d,x,y;
    scanf("%d%lf",&n,&d);
    int t=0;
    while(n||d)
    {
        t++;
        int flag=0;
        acount=0;    
        for(int i=0;i<n;i++)
        {
            scanf("%lf%lf",&x,&y);
            if(y>d){ 
                flag = 1;break;    //存在雷达无法触及的点，跳出循环，输出-1
            }

            b[i].right=x+sqrt((d*d)-(y*y));

        {

        else
            printf("Case %d: -1\n",t);
        scanf("%d%lf",&n,&d);
    }
    return 0;
}

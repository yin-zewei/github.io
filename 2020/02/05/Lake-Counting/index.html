for(int j = -1;j <= 1;++j){
int x = a + i;

dfs(x,y);
}
}
}
}

scanf(“%d%d”,&n,&m);

}
for(int i = 0;i < n;++i){
for(int j = 0;j < m;++j){

dfs(i,j);

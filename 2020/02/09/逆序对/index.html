逆序对

发表于 2020-02-09 | 分类于 算法

描述：给定一个数字序列，求满足i < j && arr[i] > arr[j]的个数

原题链接：传送门

思路：使用归并排序，根据排序原理，排序中间添加一行计数器代码，每次当左边数组的数字大于右边数组的数字的时候，说明左边数组存在m - i + 1个数(i往右边数直到中数m)大于右边数组的数，那么计数器加上m-i+1，最后计数器的数就是答案。

代码：

复杂度为O(nlogn)
```c++

#include<cstdio>
using namespace std;
#define ll long long
const int maxn = 5e5 + 5;
int arr[maxn],t[maxn]; 
ll ans = 0;
void merge(int l,int r,int m);
void msort(int l,int r){
    if(l < r){
        int m = l + (r - l)/2;
        msort(l,m);
        msort(m+1,r);
        merge(l,r,m);
    }
}
void merge(int l,int r,int m){
    int i = l,j = m + 1,id = l;
    while(i <= m && j <= r){
        if(arr[i] <= arr[j]){
            t[id++] = arr[i++];
        }else{
            ans += (ll)m - i + 1;
            t[id++] = arr[j++];
        }
    }
    while(i <= m) t[id++] = arr[i++];
    while(j <= r) t[id++] = arr[j++];
    while(l<=r) arr[l] = t[l],++l;
}
inline int read(){
    int p = 0,flag = 1;
    char c = getchar();
    while(c < '0' || c > '9'){if(c == '-') flag = -1;c = getchar();}
    while(c >= '0' && c <= '9') p = p*10 + c - '0',c = getchar();
    return p*flag;
}
int main(){
    int n = read();
    for(int i = 0;i < n;++i)arr[i] = read();
    msort(0,n - 1);
    printf("%lld",ans);
    return 0;
}

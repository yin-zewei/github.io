best cow line

Posted on 2020-02-06 | In 算法

描述
有n头奶牛，重新排序并输出，每80个换一行，使得字典序最小，但只能从头添加到新列或从尾添加到新列，以下是AC代码。

#include<cstdio>
#include<cstring>
using namespace std;
const int maxn = 100005;
int n;
char ch[maxn],ans[maxn];
//check when ch[a] == ch[b]
bool check(int a,int b){    //return 1 then from left,return 0 then from right 
    if(a == b)return 1;
    if(ch[a] == ch[b])return check(a+1,b-1);
    return ch[a] < ch[b];
}
int main(){
    scanf("%d",&n);
    for(int i = 0;i < n;++i)scanf(" %c",&ch[i]);
    int len = strlen(ch);
    int l = 0,r = len - 1,id = 0;
    while(l <= r){
        if(ch[l] == ch[r]){
            if(check(l,r)){
                ans[id] = ch[l++];
            }else{
                ans[id] = ch[r--];
            }
        }
        else if(ch[l] < ch[r]){
            ans[id] = ch[l++];
        }else{
            ans[id] = ch[r--];
        }
        //output
        putchar(ans[id++]);
        if(id % 80 == 0){
            putchar('\n');
        }
    }
}

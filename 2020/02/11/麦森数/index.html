麦森数

发表于 2020-02-11 | 分类于 算法

原题链接：传送门

描述：给定一个数P，求2^p-1的位数和后500位数字，不足500位前面补零。

思路：
前一问，直接换成10的底，位数 = log(2)*p+1
后一问，采用快速幂算法，把指数p想象成二进制数字10101…，2^p = 2^(2^j)*2^(2^k)…

代码：

```c++
#include
#include
#include
using namespace std;
int ans[1005],f[1005],tmp[1005];
//用于计算二进制P奇数位时的乘法运算
void f1(){
memset(tmp,0,sizeof(tmp));
for(int i = 1;i <= 500;++i){
for(int j = 1;j <= 500;++j){
tmp[i + j - 1] += ans[i] * f[j];//直接计算某位上的值，随后进行进位
}
}
for(int i = 1;i <= 500;++i){
tmp[i + 1] += tmp[i] / 10;
tmp[i] %= 10;
}
memcpy(ans,tmp,sizeof(tmp));
}
//用于每次对f的更新
void f2(){
memset(tmp,0,sizeof(tmp));
for(int i = 1;i <= 500;++i){
for(int j = 1;j <= 500;++j){
tmp[i + j - 1] += f[i] * f[j];
}
}
for(int i = 1;i <= 500;++i){
tmp[i + 1] += tmp[i] / 10;
tmp[i] %= 10;
}
memcpy(f,tmp,sizeof(tmp));
}
int main(){
int p;
scanf(“%d”,&p);
printf(“%d\n”,(int)(log10(2)*p+1));
f[1] = 2,ans[1] = 1;
//核心快速幂代码
while(p){
if(p&1) f1();//计算末位
p >>= 1;
f2();//更新乘数2^k
}
ans[1]-=1;
for(int i = 500;i >= 1;–i){
if(i % 50 == 0 && i != 500)putchar(‘\n’);
printf(“%d”,ans[i]);
}
return 0;
}

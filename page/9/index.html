麦森数

发表于 2020-02-11 | 分类于 题 | 阅读次数
字数统计 341 | 阅读时长 1

原题链接：传送门

描述：给定一个数P，求2^p-1的位数和后500位数字，不足500位前面补零。

思路：
前一问，直接换成10的底，位数 = log(2)*p+1
后一问，采用快速幂算法，把指数p想象成二进制数字10101…，2^p = 2^(2^j)*2^(2^k)…

代码：
```c++

#include<cstdio>
#include<cmath>
#include<cstring>
using namespace std;
int ans[1005],f[1005],tmp[1005];
//用于计算二进制P奇数位时的乘法运算
void f1(){
    memset(tmp,0,sizeof(tmp));
    for(int i = 1;i <= 500;++i){
        for(int j = 1;j <= 500;++j){
            tmp[i + j - 1] += ans[i] * f[j];//直接计算某位上的值，随后进行进位
        }
    }
    for(int i = 1;i <= 500;++i){
        tmp[i + 1] += tmp[i] / 10;
        tmp[i] %= 10;
    }
    memcpy(ans,tmp,sizeof(tmp));
}
//用于每次对f的更新
void f2(){
    memset(tmp,0,sizeof(tmp));
    for(int i = 1;i <= 500;++i){
        for(int j = 1;j <= 500;++j){
            tmp[i + j - 1] += f[i] * f[j];
        }
    }
    for(int i = 1;i <= 500;++i){
        tmp[i + 1] += tmp[i] / 10;
        tmp[i] %= 10;
    }
    memcpy(f,tmp,sizeof(tmp));
}
int main(){
    int p;
    scanf("%d",&p);
    printf("%d\n",(int)(log10(2)*p+1));
    f[1] = 2,ans[1]  = 1;
    //核心快速幂代码
    while(p){
        if(p&1) f1();//计算末位
        p >>= 1;
        f2();//更新乘数2^k
    }
    ans[1]-=1;
    for(int i = 500;i >= 1;--i){
        if(i % 50 == 0 && i != 500)putchar('\n');
        printf("%d",ans[i]);
    }
    return 0;
}

问题：a^b%c的答案是多少

原理：
如果b为偶数,a^b%c=(a*a%c)^(b/2)%c
如果b为奇数,a^b%c=(a*a%c)^(b/2)*a%c

算法：
一步步迭代，每次更新a和b，如果b为奇数，那么分离奇数项，乘进ans取余

Til the Cows Come Home

发表于 2020-02-06 | 分类于 算法 | 阅读次数
字数统计 225 | 阅读时长 1

原题链接：传送门

思路：最短路模板题，但要注意输入可能有重复路径，这里取小一点的路径
```c++

#include<cstdio>
#include<algorithm>
#include<climits>
using namespace std;
const int maxn = 1005,inf = 1e9;
int g[maxn][maxn],t,n,a,b,c,vis[maxn],dis[maxn];
int min_dis(){
    int min_val = INT_MAX,min_index;
    for(int i = 1;i <= n;++i){
        if(!vis[i] && dis[i] < min_val){
            min_val = dis[i],min_index = i; 
        }
    }
    return min_index;
}
int main(){
    fill(g[0],g[0]+maxn*maxn,inf);
    fill(vis,vis+maxn,0);
    fill(dis,dis+maxn,inf);
    scanf("%d %d",&t,&n);
    while(t--){
        scanf("%d %d %d",&a,&b,&c); 
        //坑点，需要加个判断
        if(c<g[a][b])g[a][b] = g[b][a] = c;
    }
    dis[n] = 0;
    for(int i = 1;i < n;++i){
        int u = min_dis();
        vis[u] = 1;
        for(int v = 1;v <= n;++v){
            if(!vis[v] && g[u][v] != inf && dis[u] + g[u][v] < dis[v])
                dis[v] = dis[u] + g[u][v];
        }
    }
    printf("%d",dis[1]);
    return 0; 
}

Fence Repair

发表于 2020-02-06 | 分类于 题 | 阅读次数
字数统计 248 | 阅读时长 1

来源

描述：栅栏修补，一共需要n个木条，长度不一，现要求从一个刚好够所有长度的木条上锯开，怎么锯划算，每次的花费是木条长度，如锯开21的木条，需要花费21，求最小花费。

思路：使得较长的木条使用的次数最少即可，那么从短的开始锯，每次计算两根，然后他们的和入队，重新排列。

注意：答案可能溢出，所以用long long int存储。
```c++

#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn = 20005;
int n,fence[maxn],t;
long long sum = 0;
int main(){
    scanf("%d",&n);
    //优先队列的使用
    priority_queue<int,vector<int>,greater<int>>q;
    for(int i = 0;i < n;++i){
        scanf("%d",&t);
        q.push(t);
    }
    while(q.size() > 1){
        //每次将最短的两根拿出来算
        int a = q.top();
        q.pop();
        int b = q.top();
        q.pop();
        sum += a + b;
        q.push(a+b);
    }
    printf("%lld",sum);
    return 0; 
}

快速幂取余

发表于 2020-02-09 | 分类于 算法

问题：a^b%c的答案是多少

原理：
如果b为偶数,a^b%c=(a*a%c)^(b/2)%c
如果b为奇数,a^b%c=(a*a%c)^(b/2)*a%c

算法：
一步步迭代，每次更新a和b，如果b为奇数，那么分离奇数项，乘进ans取余

Frogger

发表于 2020-02-06 | 分类于 算法

原题链接：传送门
思路：由于数据量较小，直接采用Floyd的改进算法，O(nnn)

```c++

    #include<cstdio>
    #include<algorithm>
    #include<cmath>
    using namespace std; 
    const int maxn = 205;
    int x[maxn],y[maxn],id = 1,n,a,b;
    double s[maxn][maxn];
    int main(){
        while(scanf("%d",&n) && n){
            for(int i = 1;i <= n;++i)scanf("%d %d",&x[i],&y[i]);
            for(int i = 1;i <= n;++i){
                s[i][i] = 0;
                for(int j = i + 1;j <= n;++j){
                    s[j][i] = s[i][j] = sqrt(pow(abs(x[i]-x[j])*1.0,2)+pow(abs(y[i] - y[j])*1.0,2));
                }
            }
            for(int k = 1;k <= n;++k){
                for(int i = 1;i <= n;++i){
                    for(int j = 1;j <= n;++j){
                        s[i][j] = min(s[i][j],max(s[i][k],s[k][j]));
                    }
                }
            }
            printf("Scenario #%d\nFrog Distance = %.3lf\n\n",id++,s[1][2]);;
        }
    }

Til the Cows Come Home

发表于 2020-02-06 | 分类于 算法

原题链接：传送门

思路：最短路模板题，但要注意输入可能有重复路径，这里取小一点的路径
```c++

#include<cstdio>
#include<algorithm>
#include<climits>
using namespace std;
const int maxn = 1005,inf = 1e9;
int g[maxn][maxn],t,n,a,b,c,vis[maxn],dis[maxn];
int min_dis(){
    int min_val = INT_MAX,min_index;
    for(int i = 1;i <= n;++i){
        if(!vis[i] && dis[i] < min_val){
            min_val = dis[i],min_index = i; 
        }
    }
    return min_index;
}
int main(){
    fill(g[0],g[0]+maxn*maxn,inf);
    fill(vis,vis+maxn,0);
    fill(dis,dis+maxn,inf);
    scanf("%d %d",&t,&n);
    while(t--){
        scanf("%d %d %d",&a,&b,&c); 
        //坑点，需要加个判断
        if(c<g[a][b])g[a][b] = g[b][a] = c;
    }
    dis[n] = 0;
    for(int i = 1;i < n;++i){
        int u = min_dis();
        vis[u] = 1;
        for(int v = 1;v <= n;++v){
            if(!vis[v] && g[u][v] != inf && dis[u] + g[u][v] < dis[v])
                dis[v] = dis[u] + g[u][v];
        }
    }
    printf("%d",dis[1]);
    return 0; 
}

Fence Repair

发表于 2020-02-06 | 分类于 算法

来源

描述：栅栏修补，一共需要n个木条，长度不一，现要求从一个刚好够所有长度的木条上锯开，怎么锯划算，每次的花费是木条长度，如锯开21的木条，需要花费21，求最小花费。

思路：使得较长的木条使用的次数最少即可，那么从短的开始锯，每次计算两根，然后他们的和入队，重新排列。

注意：答案可能溢出，所以用long long int存储。
```c++

#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn = 20005;
int n,fence[maxn],t;
long long sum = 0;
int main(){
    scanf("%d",&n);
    //优先队列的使用
    priority_queue<int,vector<int>,greater<int>>q;
    for(int i = 0;i < n;++i){
        scanf("%d",&t);
        q.push(t);
    }
    while(q.size() > 1){
        //每次将最短的两根拿出来算
        int a = q.top();
        q.pop();
        int b = q.top();
        q.pop();
        sum += a + b;
        q.push(a+b);
    }
    printf("%lld",sum);
    return 0; 
}

best cow line

发表于 2020-02-06 | 分类于 算法

描述
有n头奶牛，重新排序并输出，每80个换一行，使得字典序最小，但只能从头添加到新列或从尾添加到新列，以下是AC代码。
```c++

#include<cstdio>
#include<cstring>
using namespace std;
const int maxn = 100005;
int n;
char ch[maxn],ans[maxn];
//check when ch[a] == ch[b]
bool check(int a,int b){    //return 1 then from left,return 0 then from right 
    if(a == b)return 1;
    if(ch[a] == ch[b])return check(a+1,b-1);
    return ch[a] < ch[b];
}
int main(){
    scanf("%d",&n);
    for(int i = 0;i < n;++i)scanf(" %c",&ch[i]);
    int len = strlen(ch);
    int l = 0,r = len - 1,id = 0;
    while(l <= r){
        if(ch[l] == ch[r]){
            if(check(l,r)){
                ans[id] = ch[l++];
            }else{
                ans[id] = ch[r--];
            }
        }
        else if(ch[l] < ch[r]){
            ans[id] = ch[l++];
        }else{
            ans[id] = ch[r--];
        }
        //output
        putchar(ans[id++]);
        if(id % 80 == 0){
            putchar('\n');
        }
    }
}

全排列

发表于 2020-02-06 | 分类于 算法

原理

全排列是指对一个字符串或者数字或者其他任何可以排序的类型的元素数组的重新排列的所有情况

实现方法: 先把该数组排序，然后bfs搜索所有可能

c++标准库函数

在algorithm库中的next_permutation(start,end,cmp)，返回bool，中间进行对数组的变形

模板

do{
//操作
}while(next_permutation(a,a+n,cmp));

例题

描述
```c++

#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn = 15;
int n;
char ch[maxn];
char toU(char c){//转换至大写字母
    if(c >= 'a')return c - 32;
    return c;
}
bool cmp(char c1,char c2){//比较函数
    if(('a'<=c1 && c2>='a' && c1 >= 'A' && c2 <= 'Z') || abs(c1-c2) == 32)return c1 < c2;
    return toU(c1) < toU(c2);
}
int main(){
    scanf("%d",&n);
    while(n--){
        scanf("%s",ch);
        int clen = strlen(ch);
        sort(ch,ch+clen,cmp);
        do{
            printf("%s\n",ch);
        }while(next_permutation(ch,ch+clen,cmp));//全排列函数
    }
}

```c++

    #include<cstdio>
    using namespace std;
    const int maxn = 105;
    char p[maxn][maxn];
    int n,m,ans = 0;
    void dfs(int a,int b){
        p[a][b] = '.';
        for(int i = -1;i <= 1;++i){
            for(int j = -1;j <= 1;++j){
                int x = a + i;
                int y = b + j;
                if(x >= 0 && x < n && y >= 0 && y < m && p[x][y] == 'W'){
                    dfs(x,y);
                }
            }
        }
    }
    int main(){
        scanf("%d%d",&n,&m);
        for(int i = 0;i < n;++i){
            scanf("%s",p[i]);
        }
        for(int i = 0;i < n;++i){
            for(int j = 0;j < m;++j){
                if(p[i][j] == 'W'){ 
                    ans++;
                    dfs(i,j);
                } 
            }
        }
        printf("%d",ans);
        return 0;
    }

对局匹配

发表于 2020-02-01 | 分类于 算法

Description
很常见的动态规划题目，分成K组，每组用动态规划求最大不相邻子数组
```c++

#include<iostream>
using namespace std;
int book[100005],a[100005],dp[100005];
int main()
{
    int i,j,n,k,t,x=0,ans=0;
    cin>>n>>k;
    for (i=0;i<n;i++)
    {
        cin>>t;
        book[t]++;
    }
    if (k==0)
    {
        for (i=0;i<100005;i++)
        {
            if (book[i]!=0)
            ans++;
        }
    }
    else
    {
        for (i=0;i<k;i++)
        {
            x=0;
            for (j=i;j<100005;j+=k)
                a[x++]=book[j];
            for (j=0;j<x;j++)
            {
                if (j==0)
                dp[0]=a[0];
                else if (j==1)
                dp[1]=max(a[1],dp[0]);
                else
                 // 
                dp[j]=max(dp[j-1],dp[j-2]+a[j]);
            }
            ans+=dp[x-1];
        }
    }
    cout<<ans;
    return 0;
}

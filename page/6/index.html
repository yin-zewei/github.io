Til the Cows Come Home

发表于 2020-02-06 | 分类于 算法 | 阅读次数
字数统计 225 | 阅读时长 1

原题链接：传送门

思路：最短路模板题，但要注意输入可能有重复路径，这里取小一点的路径
```c++

#include<cstdio>
#include<algorithm>
#include<climits>
using namespace std;
const int maxn = 1005,inf = 1e9;
int g[maxn][maxn],t,n,a,b,c,vis[maxn],dis[maxn];
int min_dis(){
    int min_val = INT_MAX,min_index;
    for(int i = 1;i <= n;++i){
        if(!vis[i] && dis[i] < min_val){
            min_val = dis[i],min_index = i; 
        }
    }
    return min_index;
}
int main(){
    fill(g[0],g[0]+maxn*maxn,inf);
    fill(vis,vis+maxn,0);
    fill(dis,dis+maxn,inf);
    scanf("%d %d",&t,&n);
    while(t--){
        scanf("%d %d %d",&a,&b,&c); 
        //坑点，需要加个判断
        if(c<g[a][b])g[a][b] = g[b][a] = c;
    }
    dis[n] = 0;
    for(int i = 1;i < n;++i){
        int u = min_dis();
        vis[u] = 1;
        for(int v = 1;v <= n;++v){
            if(!vis[v] && g[u][v] != inf && dis[u] + g[u][v] < dis[v])
                dis[v] = dis[u] + g[u][v];
        }
    }
    printf("%d",dis[1]);
    return 0; 
}

Fence Repair

发表于 2020-02-06 | 分类于 算法 | 阅读次数
字数统计 248 | 阅读时长 1

来源

描述：栅栏修补，一共需要n个木条，长度不一，现要求从一个刚好够所有长度的木条上锯开，怎么锯划算，每次的花费是木条长度，如锯开21的木条，需要花费21，求最小花费。

思路：使得较长的木条使用的次数最少即可，那么从短的开始锯，每次计算两根，然后他们的和入队，重新排列。

注意：答案可能溢出，所以用long long int存储。
```c++

#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
const int maxn = 20005;
int n,fence[maxn],t;
long long sum = 0;
int main(){
    scanf("%d",&n);
    //优先队列的使用
    priority_queue<int,vector<int>,greater<int>>q;
    for(int i = 0;i < n;++i){
        scanf("%d",&t);
        q.push(t);
    }
    while(q.size() > 1){
        //每次将最短的两根拿出来算
        int a = q.top();
        q.pop();
        int b = q.top();
        q.pop();
        sum += a + b;
        q.push(a+b);
    }
    printf("%lld",sum);
    return 0; 
}

全排列

发表于 2020-02-06 | 分类于 算法 | 阅读次数
字数统计 308 | 阅读时长 1

原理

全排列是指对一个字符串或者数字或者其他任何可以排序的类型的元素数组的重新排列的所有情况

实现方法: 先把该数组排序，然后bfs搜索所有可能

c++标准库函数

在algorithm库中的next_permutation(start,end,cmp)，返回bool，中间进行对数组的变形

模板

do{
//操作
}while(next_permutation(a,a+n,cmp));

例题

描述
```c++

#include<cstdio>
#include<cstring>
#include<algorithm>
using namespace std;
const int maxn = 15;
int n;
char ch[maxn];
char toU(char c){//转换至大写字母
    if(c >= 'a')return c - 32;
    return c;
}
bool cmp(char c1,char c2){//比较函数
    if(('a'<=c1 && c2>='a' && c1 >= 'A' && c2 <= 'Z') || abs(c1-c2) == 32)return c1 < c2;
    return toU(c1) < toU(c2);
}
int main(){
    scanf("%d",&n);
    while(n--){
        scanf("%s",ch);
        int clen = strlen(ch);
        sort(ch,ch+clen,cmp);
        do{
            printf("%s\n",ch);
        }while(next_permutation(ch,ch+clen,cmp));//全排列函数
    }
}

    #include<cstdio>
    using namespace std;
    const int maxn = 105;
    char p[maxn][maxn];
    int n,m,ans = 0;
    void dfs(int a,int b){
        p[a][b] = '.';
        for(int i = -1;i <= 1;++i){
            for(int j = -1;j <= 1;++j){
                int x = a + i;
                int y = b + j;
                if(x >= 0 && x < n && y >= 0 && y < m && p[x][y] == 'W'){
                    dfs(x,y);
                }
            }
        }
    }
    int main(){
        scanf("%d%d",&n,&m);
        for(int i = 0;i < n;++i){
            scanf("%s",p[i]);
        }
        for(int i = 0;i < n;++i){
            for(int j = 0;j < m;++j){
                if(p[i][j] == 'W'){ 
                    ans++;
                    dfs(i,j);
                } 
            }
        }
        printf("%d",ans);
        return 0;
    }

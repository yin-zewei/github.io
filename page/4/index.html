#include<stdio.h>
#include<string>
#include<string.h>
#include<iostream>
using namespace std;
//高精度加法
//只能是两个正数相加
string add(string str1,string str2)//高精度加法
{
    string str="";
    int len1=str1.length();
    int len2=str2.length();
    //前面补0，弄成长度相同
    if(len1<len2)
    {
        for(int i=1;i<=len2-len1;i++)
           str1="0"+str1;
    }
    else
    {
        for(int i=1;i<=len1-len2;i++)
           str2="0"+str2;
    }
    len1=str1.length();
    int cf=0;
    int temp;
    for(int i=len1-1;i>=0;i--)
    {
        temp=str1[i]-'0'+str2[i]-'0'+cf;
        cf=temp/10;
        temp%=10;
        str=char(temp+'0')+str;
    }
    if(cf!=0)  str=char(cf+'0')+str;
    return str;
}
int main(){
    string a,b;
    cin >> a >> b;
    cout << add(a,b);
}

Meteor Shower

发表于 2020-02-16 | 分类于 算法

原题链接：传送门

题意：贝西在（0，0）点出发，路上有流星雨，给出流星雨的地点和降落时间，并且流星雨会摧毁该点及上下左右四个方块，每次贝西只能在1s内水平或垂直移动一个距离，求贝西是否移动到安全地点。

思路：建一个数组map[][]为图的点，记录降落时间，mark[][]为是否访问过的点，bfs遍历即可。

#include<cstdio>
#include<cstring>
#include<queue>
using namespace std;
int map[310][310];
int dir[4][2]=<!--￼0-->;
int mark[310][310];
struct node
{
    int x,y,step;
}temp,a;
bool operator < (const node &a,const node &b)
{
    return a.step>b.step;
} 

int min(int a,int b)
{
    return a>b?b:a;
}

void bfs()
{
    int i;
    memset(mark,0,sizeof(mark));
    priority_queue<node>q;
    a.x=0;
    a.y=0;
    a.step=0;
    q.push(a);
    mark[0][0]=1;
    while(!q.empty())
    {
        a=q.top();
        q.pop();
        if(map[a.x][a.y]==-1)
        {
            printf("%d\n",a.step);
            return ;
        }
        for(i=0;i<4;++i)
        {
            temp.x=a.x+dir[i][0];
            temp.y=a.y+dir[i][1];
            temp.step=a.step+1;
            if(temp.x>=0&&temp.y>=0&&(map[temp.x][temp.y]==-1||map[temp.x][temp.y]>temp.step))
            {
                if(!mark[temp.x][temp.y])
                q.push(temp);
                mark[temp.x][temp.y]=1;
            }
        }
    }
    printf("-1\n");
}



int main()
{
    int n,x,y,t,i,j;
    while(scanf("%d",&n)!=EOF)
    {
        memset(map,-1,sizeof(map));
        for(i=0;i<n;++i)
        {
            scanf("%d%d%d",&x,&y,&t);
            if(map[x][y]==-1)//判断是否受过撞击时间的更新 
                map[x][y]=t;
            else
                map[x][y]=min(map[x][y],t);//受过撞击后的点不能再走了 
            for(j=0;j<4;++j)
            {
                int nx=x+dir[j][0];
                int ny=y+dir[j][1];
                if(nx>=0&&ny>=0)
                {
                    if(map[nx][ny]==-1)
                        map[nx][ny]=t;
                    else
                        map[nx][ny]=min(t,map[nx][ny]); 
                }
            }
        }
        bfs();
    }
    return 0;

}

堆优化的Dijkstra

发表于 2020-02-14 | 分类于 算法

原题链接：传送门

#include<cstdio>
#include<algorithm>
#include<queue>
using namespace std;
typedef pair<int,int> ipair; //第一个是点，第二个是权重，
const int maxn = 5e5 + 5,inf = 0x3f3f3f3f;
vector<ipair> adj[maxn];
int dist[maxn];
int n,m,s,t1,t2,t3;
int main(){
    priority_queue<ipair,vector<ipair>,greater<ipair> > pq;
    fill(dist,dist+maxn,inf);
    scanf("%d %d %d",&n,&m,&s);
    dist[s] = 0;
    pq.push(make_pair(0,s));
    for(int i = 0;i < m;++i){
        scanf("%d %d %d",&t1,&t2,&t3);
        adj[t1].push_back(make_pair(t2,t3));
    }
    while(!pq.empty()){
        int u = pq.top().second;
        pq.pop();
        for(auto x:adj[u]){
            int v = x.first;
            int w = x.second;
            if(dist[v] > dist[u] + w){
                dist[v] = dist[u] + w;
                pq.push(make_pair(dist[v],v));
            }
        }
    }
    for(int i = 1;i <= n;++i){
        if(dist[i] != inf){
        printf("%d ",dist[i]);
        }else printf("%d ",0x7fffffff);
    }
    return 0;
}

P1378 油滴扩展

发表于 2020-02-14 | 分类于 算法

原题链接：传送门

描述：n个油滴扩展，碰到边界或者其他油滴即停止

思路：利用c++ STL自带的枚举函数，对所有点出现的顺序一一枚举，然后每次计算累加油滴的面积，即可AC

代码：

#include<cstdio>
#include<algorithm>
#include<cctype>
#include<cstring>
#include<cmath>
#define PI 3.14159265358979323846
using namespace std;
//枚举N个点，对每个点累加面积
struct point{int id,x,y;double r;}p[6];
int n,m,t[4];
double ans = 0,all;
bool cmp(point p1,point p2){return p1.id < p2.id;}
int main(){
    //初始化
    scanf("%d",&n);
    scanf("%d %d %d %d",&t[0],&t[1],&t[2],&t[3]);
    for(int i = 0;i < n;++i)scanf("%d %d",&p[i].x,&p[i].y),p[i].id = i;
    all = abs((t[0] - t[2]) * (t[1] - t[3]));
    //枚举
    do{
        //一个个的计算
        double st = 0,r = __INT_MAX__;
        for(int i = 0;i < n;++i){
            r = min(min(abs(p[i].x - t[0]),abs(p[i].x - t[2])),min(abs(p[i].y - t[1]),abs(p[i].y - t[3])));//获取由边界得到的当前点i的最小r
            for(int j = 0;j < i;++j){       //更新r
                r = min(r,sqrt(pow(p[i].x - p[j].x,2) + pow(p[i].y - p[j].y,2)) - p[j].r);
                if(r <= 0){r = 0;break;}
            }
            p[i].r = r;
            st += r*r;
        }
        ans = max(ans,st);
    }while(next_permutation(p,p+n,cmp));
    ans = ans * PI;
    printf("%d",(int)(all - ans + 0.5));
    return 0;
}

台阶问题

发表于 2020-02-13 | 分类于 算法

台阶问题是指到达第n个台阶，每部最多走k步的问题，如果dfs必然超标，只能dp
规律发现，k个以内的台阶，每次都是上面一步的两倍
k个以外的台阶，每次都是上次的两倍减去f[i-k-1]

#include<iostream>
const int mod = 100003;
int ans = 0,f[100001],n,k;
int main(){
    std::ios::sync_with_stdio(false);
    std::cin>>n>>k;
    f[0] = f[1]  = 1;
    for(int i = 2;i <= n;++i){
        if(i <= k)f[i] = 2*f[i - 1] %mod;
        else f[i] = (2*f[i-1]-f[i-k-1])%mod;
    }
    std::cout << (f[n]+mod)%mod << std::endl;
    return 0;
}
